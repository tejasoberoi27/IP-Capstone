function signIn(pageTypeValue)
{
	var pageCategory='';
	if(pageTypeValue=='HOMEPAGE'){
		pageCategory='Login';
	}
	else if(pageTypeValue=='cartSummaryPage'){
		pageCategory='Single page checkout';
	}
	else if(pageTypeValue=='CART'){
		pageCategory='Single page checkout';
	}
	
	ga('send', {
        hitType: 'event',
        eventCategory: pageCategory,
        eventAction: 'successful login as returning customer'
      });
	
	if(pageTypeValue == 'DELIVERYMETHODCHECKOUT')
	{	
		var productCodes=$("#productCode").val().split(',');
		var productCode=$("#productCode").val();
		var productCodesWithComma = productCode.trim();
		var productName=$("#productName").val().split('|');

		var productCategory = $("#productcategory").val().split(',');

		var quantity = $("#producquantity").val().split(',');
		
		
		productCodes.forEach(function(productCodes, index) {
			step2TagGA(productCodes.trim(), productCategory[index].trim(), manufacturerName[index].trim(), productName[index].trim(), quantity[index].trim(),productPrices[index])
       	});    
	       ga('ec:setAction','checkout', {'step': 2});

           ga('send', 'event', 'Single page checkout',productCodesWithComma, 2);

	{

	       ga('ec:addProduct',{
	              id: productcode,
	              name: productname,
	              category: category,
